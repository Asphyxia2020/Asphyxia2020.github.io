5. 欧几里得算法求最大公约数和最小公倍数    

```C++
// 最大公约数
int Fraction::gcd(int x, int y) const {
    // 欧几里得算法
    while (y) {
        int t = y;
        y = x % y;
        x = t;
    }
    return x;
}

// 最小公倍数
int Fraction::lcm(int x, int y) const {
    int n = gcd(x, y);
    return x * y / n;
} 
```
